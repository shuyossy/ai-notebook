import 'dotenv/config';
import { NodeSDK } from '@opentelemetry/sdk-node';
import { getNodeAutoInstrumentations } from '@opentelemetry/auto-instrumentations-node';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes as SRA } from '@opentelemetry/semantic-conventions';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-http';

const resource = new Resource({
  [SRA.SERVICE_NAME]: process.env.OTEL_SERVICE_NAME ?? 'ai-notebook',
});

const auth = Buffer.from(
  `${process.env.LANGFUSE_PUBLIC_KEY}:${process.env.LANGFUSE_SECRET_KEY}`
).toString('base64');

const traceExporter = new OTLPTraceExporter({
  url: `${process.env.LANGFUSE_BASE_URL ?? 'https://cloud.langfuse.com'}/api/public/otel/v1/traces`,
  headers: { Authorization: `Basic ${auth}` },
});

export const sdk = new NodeSDK({
  resource,
  traceExporter,
  instrumentations: [getNodeAutoInstrumentations()],
});

sdk.start();
